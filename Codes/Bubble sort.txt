public class MyClass {
    public static void main(String args[]) {
    int arr[] = {3,5,2,7,1,6};
    int temp;
    for(int i=0; i<arr.length-1; i++){
        for(int j=0; j<arr.length-i-1; j++){
            if(arr[j]>arr[j+1]){
                temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
            }
        }
    }
    
    for(int i:arr){
        System.out.println(i);
    }
    
    }
}
